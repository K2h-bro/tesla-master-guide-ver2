import express from "express"; import Stripe from "stripe"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); const stripe = new Stripe("여기에_STRIPE_SECRET_KEY"); app.post("/api/checkout", async (req, res) => { const items = req.body.items; const line_items = items.map(i => ({ price_data: { currency: "krw", product_data: { name: i.name }, unit_amount: i.price }, quantity: 1 })); const session = await stripe.checkout.sessions.create({ payment_method_types: ["card"], mode: "payment", line_items, success_url: "http://localhost:5500/success", cancel_url: "http://localhost:5500/cancel" }); res.json({ url: session.url }); }); app.listen(3000, ()=>console.log("Server running"));
